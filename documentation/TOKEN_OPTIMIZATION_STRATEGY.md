```typescript
// src/lib/claude-client.ts

interface CacheablePrompt {
  system: string;           // System instructions (cacheable)
  context: string;          // Code context, docs (cacheable)
  userMessage: string;      // User's actual question (not cached)
}

async function callClaudeWithCache(prompt: CacheablePrompt) {
  return await anthropic.messages.create({
    model: "claude-sonnet-4-20250514",
    max_tokens: 4096,
    system: [
      {
        type: "text",
        text: prompt.system,
        cache_control: { type: "ephemeral" }  // Cache this!
      }
    ],
    messages: [
      {
        role: "user",
        content: [
          {
            type: "text",
            text: prompt.context,
            cache_control: { type: "ephemeral" }  // Cache this too!
          },
          {
            type: "text",
            text: prompt.userMessage  // This changes per request
          }
        ]
      }
    ]
  });
}
```

### When to Use Batch API in QUAD

| Use Case | Batch? | Why |
|----------|--------|-----|
| Interactive coding chat | NO | Need immediate response |
| Code review (multiple files) | YES | Can wait, bulk operation |
| Generate documentation | YES | Async task, can wait |
| Analyze codebase | YES | Background job |
| Meeting transcript â†’ tickets | YES | Async processing |
| Daily health reports | YES | Scheduled, non-urgent |

// src/lib/batch-processor.ts

async function submitBatchJob(requests: BatchRequest[]) {
  // Create JSONL file with requests
  const jsonlContent = requests.map(r => JSON.stringify({
    custom_id: r.id,
    params: {
      model: "claude-sonnet-4-20250514",
      max_tokens: 4096,
      messages: r.messages
    }
  })).join('\n');

  // Upload and create batch
  const batch = await anthropic.batches.create({
    requests: jsonlContent
  });

  return batch.id;  // Poll this for results
}

  messages: [{ role: 'user', content: `Review this code:\n${readFile(file)}` }]

## Checking Your Token Usage

To check your current Claude API usage:

1. **Console**: Visit https://console.anthropic.com/settings/usage
2. **API**: Call the usage endpoint with your API key
3. **Claude Code**: Check `~/.claude/` for local usage logs

Note: Claude Code (the CLI tool) tracks usage separately from API usage. Your CLI usage goes through Anthropic's consumer infrastructure.

## Summary

| Strategy | Effort | Savings | When to Use |
|----------|--------|---------|-------------|
| Prompt Caching | Low | 50-90% on input | Always - easy win |
| Model Routing | Medium | 30-50% | Most tasks |
| Context Compression | Medium | 20-40% | Long conversations |
| Batch API | Medium | 50% | Async/bulk operations |
| Combined | High | 50-70% total | Full optimization |

**Key Takeaway**: With 1M tokens and proper optimization:
- Without optimization: ~140 average requests
- With full optimization: ~350+ requests (2.5x more!)
